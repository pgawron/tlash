\resetsteps      % Reset all the commands to create a blank worksheet  

% Define the operation to be computed

\renewcommand{\operation}{ C := A^T B^T + C }

% Step 1a: Precondition 

\renewcommand{\precondition}{
  C = \hat{C}
}

% Step 1b: Postcondition 

\renewcommand{\postcondition}{ 
  C
  =
  A^T B^T + \hat{C}
}

% Step 2: Invariant 
% Note: Right-hand side of equalities must be updated appropriately

\renewcommand{\invariant}{
  \FlaTwoByOne{C_T}
              {C_B} = 
  \FlaTwoByOne{ A_{L}^T B^T + \hat{C}_T }
              {               \hat{C}_B }
}

% Step 3: Loop-guard 

\renewcommand{\guard}{
  n( A_R ) < n( A )
}

% Step 4: Initialize 

\renewcommand{\partitionings}{
  $
  A \rightarrow
  \FlaOneByTwo{A_L}{A_R}
  $
,
  $
  C \rightarrow
  \FlaTwoByOne{C_{T}}
              {C_{B}}
  $
,
\\ 
  $
  \hat{C} \rightarrow
  \FlaTwoByOne{\hat{C}_{T}}
              {\hat{C}_{B}}
  $
}

\renewcommand{\partitionsizes}{
$ A_R $ has $ 0 $ columns,
$ C_{B} $ has $ 0 $ rows,
$ \hat{C}_{B} $ has $ 0 $ rows
}

% Step 5a: Repartition the operands 

\renewcommand{\blocksize}{b}

\renewcommand{\repartitionings}{
$
  \FlaOneByTwo{A_L}{A_R}
\rightarrow  \FlaOneByThreeL{A_0}{A_1}{A_2}
$
,
$
  \FlaTwoByOne{ C_T }
              { C_B }
\rightarrow
  \FlaThreeByOneT{C_0}
                 {C_1}
                 {C_2}
$
,
\\ 
$
  \FlaTwoByOne{ \hat{C}_T }
              { \hat{C}_B }
\rightarrow
  \FlaThreeByOneT{\hat{C}_0}
                 {\hat{C}_1}
                 {\hat{C}_2}
$
}

\renewcommand{\repartitionsizes}{
$ A_1 $ has $b$ columns,
$ C_1 $ has $ b $ rows,
$ \hat{C}_1 $ has $ b $ rows}

% Step 5b: Move the double lines 

\renewcommand{\moveboundaries}{
$
  \FlaOneByTwo{A_L}{A_R}
\leftarrow  \FlaOneByThreeR{A_0}{A_1}{A_2}
$
,
$
  \FlaTwoByOne{ C_T }
              { C_B }
\leftarrow
  \FlaThreeByOneB{C_0}
                 {C_1}
                 {C_2}
$
,
\\ 
$
  \FlaTwoByOne{ \hat{C}_T }
              { \hat{C}_B }
\leftarrow
  \FlaThreeByOneB{\hat{C}_0}
                 {\hat{C}_1}
                 {\hat{C}_2}
$
}

% Step 6: State after repartitioning
% Note: The below needs editing!!!

\renewcommand{\beforeupdate}{
  \FlaThreeByOneT{C_0}
                 {C_1}
                 {C_2}

=
  \FlaThreeByOneT{               \hat{C}_0 }
                 {               \hat{C}_1 }
                 { A_{2}^T B^T + \hat{C}_2 }

}

% Step 7: State after moving of double lines
% Note: The below needs editing!!!

\renewcommand{\afterupdate}{
  \FlaThreeByOneT{C_0}
                 {C_1}
                 {C_2}

=
  \FlaThreeByOneT{               \hat{C}_0 }
                 { A_{1}^T B^T + \hat{C}_1 }
                 { A_{2}^T B^T + \hat{C}_2 }

}

% Step 8: Insert the updates required to change the 
%         state from that given in Step 6 to that given in Step 7
% Note: The below needs editing!!!

\renewcommand{\update}{
$
  \begin{array}{l}
    C_{1} := A_{1}^T B^T + \hat{C}_1 \\
  \end{array}
$
}
