\resetsteps      % Reset all the commands to create a blank worksheet  

% Define the operation to be computed

\renewcommand{\operation}{ C := A^T B + C }

% Step 1a: Precondition 

\renewcommand{\precondition}{
  C = \hat{C}
}

% Step 1b: Postcondition 

\renewcommand{\postcondition}{ 
  C
  =
  A^T B + \hat{C}
}

% Step 2: Invariant 
% Note: Right-hand side of equalities must be updated appropriately

\renewcommand{\invariant}{
  \FlaOneByTwo{C_L}{C_R}  = 
  \FlaOneByTwo{\hat{C}_L}{ A^T B_{R} + \hat{C}_R } 
}

% Step 3: Loop-guard 

\renewcommand{\guard}{
  n( B_R ) < n( B )
}

% Step 4: Initialize 

\renewcommand{\partitionings}{
  $
  B \rightarrow
  \FlaOneByTwo{B_L}{B_R}
  $
,
  $
  C \rightarrow
  \FlaOneByTwo{C_L}{C_R}
  $
,
\\ 
  $
  \hat{C} \rightarrow
  \FlaOneByTwo{\hat{C}_L}{\hat{C}_R}
  $
}

\renewcommand{\partitionsizes}{
$ B_R $ has $ 0 $ columns,
$ C_R $ has $ 0 $ columns,
$ \hat{C}_R $ has $ 0 $ columns
}

% Step 5a: Repartition the operands 

\renewcommand{\blocksize}{b}

\renewcommand{\repartitionings}{
$
  \FlaOneByTwo{B_L}{B_R}
\rightarrow  \FlaOneByThreeL{B_0}{B_1}{B_2}
$
,
$
  \FlaOneByTwo{C_L}{C_R}
\rightarrow  \FlaOneByThreeL{C_0}{C_1}{C_2}
$
,
\\ 
$
  \FlaOneByTwo{\hat{C}_L}{\hat{C}_R}
\rightarrow  \FlaOneByThreeL{\hat{C}_0}{\hat{C}_1}{\hat{C}_2}
$
}

\renewcommand{\repartitionsizes}{
$ B_1 $ has $b$ columns,
$ C_1 $ has $b$ columns,
$ \hat{C}_1 $ has $b$ columns}

% Step 5b: Move the double lines 

\renewcommand{\moveboundaries}{
$
  \FlaOneByTwo{B_L}{B_R}
\leftarrow  \FlaOneByThreeR{B_0}{B_1}{B_2}
$
,
$
  \FlaOneByTwo{C_L}{C_R}
\leftarrow  \FlaOneByThreeR{C_0}{C_1}{C_2}
$
,
\\ 
$
  \FlaOneByTwo{\hat{C}_L}{\hat{C}_R}
\leftarrow  \FlaOneByThreeR{\hat{C}_0}{\hat{C}_1}{\hat{C}_2}
$
}

% Step 6: State after repartitioning
% Note: The below needs editing!!!

\renewcommand{\beforeupdate}{
  \FlaOneByThreeL{C_0}{C_1}{C_2}
=
  \FlaOneByThreeL{\hat{C}_0}{\hat{C}_1}{ A^T B_{2} + \hat{C}_2 }
}

% Step 7: State after moving of double lines
% Note: The below needs editing!!!

\renewcommand{\afterupdate}{
  \FlaOneByThreeR{C_0}{C_1}{C_2}
=
  \FlaOneByThreeR{\hat{C}_0}{ A^T B_{1} + \hat{C}_1 }{ A^T B_{2} + \hat{C}_2 }
}

% Step 8: Insert the updates required to change the 
%         state from that given in Step 6 to that given in Step 7
% Note: The below needs editing!!!

\renewcommand{\update}{
$
  \begin{array}{l}
    C_{1} := A^T B_{1} + \hat{C}_1 \\
  \end{array}
$
}
